     if [ -s "/etc/inotifywait.log" ];then

        rsync -az --delete --exclude="*.swp" --exclude="*.swx" $watch_dir $push_to:/tmp

           echo "$watch_dir sync to $push_to failed at `date +"%F %T"`,please check it by manual" |\

        cat /dev/null > /etc/inotifywait.log
        rsync -az --delete --exclude="*.swp" --exclude="*.swx" $watch_dir $push_to:/tmp
    else
        sleep 1
    fi
